## Snake and Ladder Game ‚Äì LLD
### Core Requirements
- N x N board.
- Snakes move players back.
- Ladders move players forward.
- Dice rolls 1‚Äì6.
- Multiple players take turns.
- First to reach the final cell wins.

```go
package main

import (
	"fmt"
	"math/rand"
	"time"
)

// ---------------------------
// Player
// ---------------------------

type Player struct {
	name     string
	position int
}

// ---------------------------
// Board
// ---------------------------

type Board struct {
	size    int
	snakes  map[int]int
	ladders map[int]int
}

func (b *Board) GetNextPosition(pos int) int {
	if end, ok := b.snakes[pos]; ok {
		fmt.Printf("üêç Bitten by a snake! Slide from %d to %d\n", pos, end)
		return end
	}
	if end, ok := b.ladders[pos]; ok {
		fmt.Printf("ü™ú Climbed a ladder! Climb from %d to %d\n", pos, end)
		return end
	}
	return pos
}

// ---------------------------
// Dice
// ---------------------------

type Dice struct{}

func (d *Dice) Roll() int {
	return rand.Intn(6) + 1
}

// ---------------------------
// Singleton Config
// ---------------------------

type GameConfig struct {
	board *Board
	dice  *Dice
}

var config *GameConfig

func GetGameConfig() *GameConfig {
	if config == nil {
		config = &GameConfig{
			board: &Board{
				size: 100,
				snakes: map[int]int{
					99: 54, 70: 55, 52: 42,
				},
				ladders: map[int]int{
					2: 38, 15: 26, 8: 31,
				},
			},
			dice: &Dice{},
		}
	}
	return config
}

// ---------------------------
// Command Pattern
// ---------------------------

type Command interface {
	Execute()
}

type MoveCommand struct {
	player *Player
	dice   *Dice
	board  *Board
}

func (cmd *MoveCommand) Execute() {
	roll := cmd.dice.Roll()
	fmt.Printf("üé≤ %s rolled a %d\n", cmd.player.name, roll)

	next := cmd.player.position + roll
	if next > cmd.board.size {
		fmt.Printf("üö´ %s can't move. Still at %d\n", cmd.player.name, cmd.player.position)
		return
	}

	next = cmd.board.GetNextPosition(next)
	cmd.player.position = next
	fmt.Printf("‚û°Ô∏è %s is now at position %d\n\n", cmd.player.name, cmd.player.position)
}

// ---------------------------
// Game Loop
// ---------------------------

func main() {
	rand.Seed(time.Now().UnixNano())
	cfg := GetGameConfig()

	players := []*Player{
		{name: "Alice"},
		{name: "Bob"},
	}
	turnQueue := []int{0, 1}
	gameOver := false

	for !gameOver {
		currentIndex := turnQueue[0]
		turnQueue = turnQueue[1:]

		player := players[currentIndex]
		cmd := &MoveCommand{
			player: player,
			dice:   cfg.dice,
			board:  cfg.board,
		}
		cmd.Execute()

		if player.position == cfg.board.size {
			fmt.Printf("üèÅ %s wins the game!\n", player.name)
			gameOver = true
		} else {
			turnQueue = append(turnQueue, currentIndex)
		}
	}
}
```
