```golang
package main

import (
	"container/list"
	"fmt"
	"sync"
	"time"
)

///////////////////////////////////////////////////////////
// 🔹 Entry
///////////////////////////////////////////////////////////

type entry[K comparable, V any] struct {
	key      K
	value    V
	expireAt time.Time
}

///////////////////////////////////////////////////////////
// 🔹 Cache
///////////////////////////////////////////////////////////

type Cache[K comparable, V any] struct {
	capacity int
	items    map[K]*list.Element
	ll       *list.List

	mu sync.RWMutex

	stopJanitor chan struct{}
}

///////////////////////////////////////////////////////////
// 🔹 Constructor
///////////////////////////////////////////////////////////

func NewCache[K comparable, V any](capacity int, cleanupInterval time.Duration) *Cache[K, V] {
	c := &Cache[K, V]{
		capacity:    capacity,
		items:       make(map[K]*list.Element),
		ll:          list.New(),
		stopJanitor: make(chan struct{}),
	}

	go c.startJanitor(cleanupInterval)
	return c
}

///////////////////////////////////////////////////////////
// 🔹 Public APIs
///////////////////////////////////////////////////////////

func (c *Cache[K, V]) Set(key K, value V, ttl time.Duration) {
	c.mu.Lock()
	defer c.mu.Unlock()

	// If exists → update + move to front
	if elem, ok := c.items[key]; ok {
		e := elem.Value.(*entry[K, V])
		e.value = value
		e.expireAt = time.Now().Add(ttl)

		c.ll.MoveToFront(elem)
		return
	}

	// New entry
	e := &entry[K, V]{
		key:      key,
		value:    value,
		expireAt: time.Now().Add(ttl),
	}

	elem := c.ll.PushFront(e)
	c.items[key] = elem

	// Evict if needed
	if c.ll.Len() > c.capacity {
		c.evictLRU()
	}
}

func (c *Cache[K, V]) Get(key K) (V, bool) {
	c.mu.Lock() // write lock because we mutate LRU order
	defer c.mu.Unlock()

	elem, ok := c.items[key]
	if !ok {
		var zero V
		return zero, false
	}

	e := elem.Value.(*entry[K, V])

	// TTL check (lazy expiration)
	if time.Now().After(e.expireAt) {
		c.removeElement(elem)
		var zero V
		return zero, false
	}

	// Move to front (LRU)
	c.ll.MoveToFront(elem)
	return e.value, true
}

func (c *Cache[K, V]) Delete(key K) {
	c.mu.Lock()
	defer c.mu.Unlock()

	if elem, ok := c.items[key]; ok {
		c.removeElement(elem)
	}
}

///////////////////////////////////////////////////////////
// 🔹 Internal Helpers
///////////////////////////////////////////////////////////

func (c *Cache[K, V]) evictLRU() {
	elem := c.ll.Back()
	if elem == nil {
		return
	}
	c.removeElement(elem)
}

func (c *Cache[K, V]) removeElement(elem *list.Element) {
	e := elem.Value.(*entry[K, V])
	delete(c.items, e.key)
	c.ll.Remove(elem)
}

///////////////////////////////////////////////////////////
// 🔹 Janitor (Background Cleanup)
///////////////////////////////////////////////////////////

func (c *Cache[K, V]) startJanitor(interval time.Duration) {
	ticker := time.NewTicker(interval)

	for {
		select {
		case <-ticker.C:
			c.cleanup()
		case <-c.stopJanitor:
			ticker.Stop()
			return
		}
	}
}

func (c *Cache[K, V]) cleanup() {
	c.mu.Lock()
	defer c.mu.Unlock()

	now := time.Now()

	// Iterate from LRU side (older items more likely expired)
	for elem := c.ll.Back(); elem != nil; {
		prev := elem.Prev()

		e := elem.Value.(*entry[K, V])
		if now.After(e.expireAt) {
			c.removeElement(elem)
		}

		elem = prev
	}
}

func (c *Cache[K, V]) Stop() {
	close(c.stopJanitor)
}

///////////////////////////////////////////////////////////
// 🔹 Example Usage
///////////////////////////////////////////////////////////

func main() {
	cache := NewCache[string, string](2, 2*time.Second)
	defer cache.Stop()

	cache.Set("a", "apple", 3*time.Second)
	cache.Set("b", "banana", 5*time.Second)

	// Access
	if v, ok := cache.Get("a"); ok {
		fmt.Println("a:", v)
	}

	// Wait for TTL expiration
	time.Sleep(4 * time.Second)

	if _, ok := cache.Get("a"); !ok {
		fmt.Println("a expired")
	}

	// Trigger LRU eviction
	cache.Set("c", "cherry", 5*time.Second)

	if _, ok := cache.Get("b"); !ok {
		fmt.Println("b evicted (LRU)")
	} else {
		fmt.Println("b still exists")
	}
}
```

```python
import threading
import time

class CacheItem:
    def __init__(self, value, ttl):
        self.value = value
        self.expiration = time.time() + ttl

class TTLCache:
    def __init__(self, ttl):
        self.ttl = ttl
        self.store = {}
        self.lock = threading.Lock()
        threading.Thread(target=self._cleanup_loop, daemon=True).start()

    def set(self, key, value):
        with self.lock:
            self.store[key] = CacheItem(value, self.ttl)

    def get(self, key):
        with self.lock:
            item = self.store.get(key)
            if not item or time.time() > item.expiration:
                return None
            return item.value

    def _cleanup_loop(self):
        while True:
            time.sleep(self.ttl / 2)
            with self.lock:
                now = time.time()
                keys_to_delete = [k for k, v in self.store.items() if now > v.expiration]
                for k in keys_to_delete:
                    del self.store[k]

# Example usage
if __name__ == "__main__":
    cache = TTLCache(ttl=5)
    cache.set("foo", "bar")
    print(cache.get("foo"))  # bar
    time.sleep(6)
    print(cache.get("foo"))  # None
```

```golang
package main

import (
	"fmt"
	"sync"
	"time"
)

type CacheItem struct {
	value      interface{}
	expiration time.Time
}

type TTLCache struct {
	items map[string]CacheItem
	mu    sync.RWMutex
	ttl   time.Duration
}

// Constructor
func NewTTLCache(ttl time.Duration) *TTLCache {
	cache := &TTLCache{
		items: make(map[string]CacheItem),
		ttl:   ttl,
	}
	// Start cleanup goroutine
	go cache.cleanupLoop()
	return cache
}

// Set a value
func (c *TTLCache) Set(key string, value interface{}) {
	c.mu.Lock()
	defer c.mu.Unlock()
	c.items[key] = CacheItem{
		value:      value,
		expiration: time.Now().Add(c.ttl),
	}
}

// Get a value
func (c *TTLCache) Get(key string) (interface{}, bool) {
	c.mu.RLock()
	defer c.mu.RUnlock()
	item, found := c.items[key]
	if !found {
		return nil, false
	}
	if time.Now().After(item.expiration) {
		return nil, false
	}
	return item.value, true
}

// Background cleanup to remove expired items
func (c *TTLCache) cleanupLoop() {
	for {
		time.Sleep(c.ttl / 2)
		c.mu.Lock()
		now := time.Now()
		for k, v := range c.items {
			if now.After(v.expiration) {
				delete(c.items, k)
			}
		}
		c.mu.Unlock()
	}
}

// Example usage
func main() {
	cache := NewTTLCache(5 * time.Second)

	cache.Set("foo", "bar")
	val, ok := cache.Get("foo")
	fmt.Println("Before expiry:", val, ok) // bar true

	time.Sleep(6 * time.Second)

	val, ok = cache.Get("foo")
	fmt.Println("After expiry:", val, ok) // <nil> false
}
```

```golang
// TTL Cache in Go (non-generic) + Min-Heap-based eviction
package main

import (
	"container/heap"
	"fmt"
	"sync"
	"time"
)

type cacheEntry struct {
	key       string
	value     interface{}
	expiresAt time.Time
	index     int // for heap updates
}

type TTLCache struct {
	data     map[string]*cacheEntry
	heap     priorityQueue
	duration time.Duration
	mutex    sync.Mutex
}

func NewTTLCache(ttl time.Duration) *TTLCache {
	c := &TTLCache{
		data:     make(map[string]*cacheEntry),
		heap:     make(priorityQueue, 0),
		duration: ttl,
	}
	go c.startEvictionLoop()
	return c
}

func (c *TTLCache) Put(key string, value interface{}) {
	c.mutex.Lock()
	defer c.mutex.Unlock()

	if existing, ok := c.data[key]; ok {
		existing.value = value
		existing.expiresAt = time.Now().Add(c.duration)
		heap.Fix(&c.heap, existing.index)
		return
	}

	ce := &cacheEntry{
		key:       key,
		value:     value,
		expiresAt: time.Now().Add(c.duration),
	}
	heap.Push(&c.heap, ce)
	c.data[key] = ce
}

func (c *TTLCache) Get(key string) (interface{}, bool) {
	c.mutex.Lock()
	defer c.mutex.Unlock()
	if entry, ok := c.data[key]; ok && time.Now().Before(entry.expiresAt) {
		return entry.value, true
	}
	return nil, false
}

func (c *TTLCache) startEvictionLoop() {
	for {
		time.Sleep(time.Second)
		c.evictExpired()
	}
}

func (c *TTLCache) evictExpired() {
	c.mutex.Lock()
	defer c.mutex.Unlock()
	now := time.Now()
	for c.heap.Len() > 0 {
		min := c.heap[0]
		if now.Before(min.expiresAt) {
			break
		}
		heap.Pop(&c.heap)
		delete(c.data, min.key)
	}
}

// Priority queue (min-heap) based on expiration time
type priorityQueue []*cacheEntry

func (pq priorityQueue) Len() int { return len(pq) }

func (pq priorityQueue) Less(i, j int) bool {
	return pq[i].expiresAt.Before(pq[j].expiresAt)
}

func (pq priorityQueue) Swap(i, j int) {
	pq[i], pq[j] = pq[j], pq[i]
	pq[i].index = i
	pq[j].index = j
}

func (pq *priorityQueue) Push(x interface{}) {
	n := len(*pq)
	entry := x.(*cacheEntry)
	entry.index = n
	*pq = append(*pq, entry)
}

func (pq *priorityQueue) Pop() interface{} {
	old := *pq
	n := len(old)
	entry := old[n-1]
	entry.index = -1
	*pq = old[0 : n-1]
	return entry
}

func main() {
	cache := NewTTLCache(3 * time.Second)
	cache.Put("a", 1)
	cache.Put("b", 2)
	fmt.Println(cache.Get("a"))
	time.Sleep(4 * time.Second)
	fmt.Println(cache.Get("a"))
}
```
